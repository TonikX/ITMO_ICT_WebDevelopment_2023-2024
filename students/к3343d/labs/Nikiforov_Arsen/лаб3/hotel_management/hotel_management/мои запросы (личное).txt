получить список клиентов по комнатам

from hotel_api.models import Room, Client
# Получение всех комнат с информацией о клиентах
rooms_with_clients = Room.objects.prefetch_related('clients').all()
for room in rooms_with_clients:
    print(f"Комната: {room.room_type} на этаже {room.floor.number}")
    clients_in_room = room.clients.all()
    if clients_in_room:
        print("Клиенты в этой комнате:")
        for client in clients_in_room:
            print(f"{client.client_info.first_name} {client.client_info.last_name}")
    else:
        print("В этой комнате нет клиентов.")
    print("-" * 30)
